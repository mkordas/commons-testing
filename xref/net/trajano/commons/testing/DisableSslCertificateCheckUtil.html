23   * Disables SSL certificate checks. Primarily used for doing integration testing
24   * against self-signed servers.
25   */
26  public final class DisableSslCertificateCheckUtil {
27      /**
28       * Flag to indicate that certificate checks are disabled. If this is true,
29       * then the process to disable the checks are not executed again.
30       */
31      private static boolean disabled;
32  
33      /**
34       * Logger.
35       */
36      private static final Logger LOG = Logger.getLogger(
37              DisableSslCertificateCheckUtil.class.getName(),
38              "net.trajano.commons.testing.Messages");

40      /**
41       * Original hostname verifier, set by {{@link #disableChecks()}.
42       */
43      private static HostnameVerifier originalHostnameVerifier;
44  
45      /**
46       * Original SSL Socket factory, set by {{@link #disableChecks()}.
47       */
48      private static SSLSocketFactory originalSslSocketFactory;
49  
50      /**
51       * Disable trust checks for SSL connections. Saves the present ones if it is
52       * not the disabled ones.
53       * 
54       * @throws GeneralSecurityException
55       *             thrown when there is a problem disabling the SSL. Shouldn't
56       *             happen unless there is something wrong with the Java
57       *             implementation.
58       */
59      public static void disableChecks() throws GeneralSecurityException {
60          if (disabled) {
61              return;
62          }
63          try {
64              new URL("https", "0", "/").getContent();
65          } catch (final IOException e) {
66              // This invocation will always fail, but it will register the
67              // default SSL provider to the URL class.
68              LOG.log(Level.FINEST,
69                      "DisableSSLCertificateCheckUtil.disableCertificateCheck", e);
70          }
71          originalSslSocketFactory = getDefaultSSLSocketFactory();
72          originalHostnameVerifier = getDefaultHostnameVerifier();
73          final SSLContext context = SSLContext.getInstance("SSLv3");
74          final TrustManager[] trustManagerArray = { new NullX509TrustManager() };
75          context.init(null, trustManagerArray, null);
76          setDefaultSSLSocketFactory(context.getSocketFactory());
77          setDefaultHostnameVerifier(new NullHostnameVerifier());
78          disabled = true;
79      }
80  
81      /**
82       * This will re-enable the SSL checks after it was disabled by
83       * {@link #disableChecks()}.
84       */
85      public static void reenableChecks() {
86          if (!disabled) {
87              return;
88          }
89          setDefaultSSLSocketFactory(originalSslSocketFactory);
90          setDefaultHostnameVerifier(originalHostnameVerifier);
91          disabled = false;
92      }
93  
94      /**
95       * Prevent instantiation of utility class.
96       */
97      private DisableSslCertificateCheckUtil() {
98      }
